CREATE PROC USP_GETEMPLOYEES
AS
BEGIN
	SELECT ROW_NUMBER() OVER(ORDER BY NAME) AS ROWNUMBER, 
	NAME,DESIGNATION,SALARY,STATE,COUNTRY FROM DBO.EMPLOYEEMASTER
	ORDER BY NAME
END
